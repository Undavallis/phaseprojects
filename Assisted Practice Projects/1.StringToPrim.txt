public class StringToPrim 
    {

	public static void main(String[] args) 
	{
	     String s="100";  //Declaring String variable  
	     int i=Integer.parseInt(s);  //Converting String into int using Integer.parseInt()  
             System.out.println(i);  //Printing value of i
             
             float j=Float.parseFloat(s);  //Converting String into float using Float.parseInt()  
             System.out.println(j);  //Printing value of j
             
             double k=Double.parseDouble(s);  //Converting String into double using Double.parseInt()  
             System.out.println(k);  //Printing value of k
	} 
	}
